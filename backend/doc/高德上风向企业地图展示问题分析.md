  🗺️ 地图渲染组件

  1. SDK加载 Hook（useAMapLoader.ts）

  // 动态加载AMap脚本
  const script = document.createElement('script')
  script.src = `https://webapi.amap.com/maps?v=2.0&key=${amapKey}`
  script.onload = () => setLoaded(true)
  document.body.appendChild(script)

  return { loaded, AMap: window.AMap }

  2. 地图面板组件（MapPanel.tsx 第14-95行）

  const MapPanel: React.FC<Props> = ({ payload, amapKey }) => {
      const { loaded, AMap } = useAMapLoader(amapKey)

      useEffect(() => {
          // 1. 检查条件
          if (!amapKey || !loaded || !AMap) {
              console.log('地图加载条件不满足')
              return
          }

          // 2. 初始化地图
          const map = new AMap.Map(mapContainerRef.current, {
              zoom: 11,
              center: [center.lng, center.lat]
          })

          // 3. 添加站点标记（红色）
          if (payload.station) {
              const stationMarker = new AMap.Marker({
                  position: [payload.station.lng, payload.station.lat],
                  title: payload.station.name,
                  icon: new AMap.Icon({
                      image: '//a.amap.com/jsapi_demos/static/demo-center/icons/poi-marker-red.png'
                  })
              })
              map.add(stationMarker)
          }

          // 4. 添加企业标记（灰色）
          if (payload.enterprises) {
              const markers = payload.enterprises.map(ent =>
                  new AMap.Marker({
                      position: [ent.lng, ent.lat],
                      title: `${ent.name} (${ent.industry})`
                  })
              )
              map.add(markers)
          }

          // 5. 添加上风向路径（绿色）
          if (payload.upwind_paths) {
              const paths = payload.upwind_paths.map(p =>
                  new AMap.Polyline({
                      path: p.points.map(pt => [pt.lng, pt.lat]),
                      strokeColor: '#39ff14'
                  })
              )
              map.add(paths)
          }

      }, [payload, amapKey, loaded, AMap])

      // 未配置Key
      if (!amapKey) {
          return <div>AMap Key未配置，无法加载动态地图。</div>
      }

      // 加载中
      if (!loaded) {
          return <div>地图加载中…</div>
      }

      return <div ref={mapContainerRef} style={{ height: '400px' }} />
  }

  ⚠️ 当前问题分析

  问题现象

  高德地图无法显示，即使Key已配置（337ddd852a2ec4b42aa3442729a4026a）

  可能原因

  1. 地图数据未正确传递到 weather_analysis 模块

  当前逻辑：
  - weather_analysis.visuals = []（空数组）
  - 地图数据在 upwind_enterprises 字段
  - 前端在 onDone 时手动构建地图payload

  问题：
  - 全屏模式使用 moduleData.weather_analysis
  - 但 weather_analysis.visuals 是空的
  - 所以全屏模式中看不到地图

  2. 对话框模式可能能看到地图

  对话框模式通过 onDone 处理 upwind_enterprises：
  // App.tsx 第145-181行
  if (data.data?.upwind_enterprises) {
      // 手动构建mapPayload并添加到messages
      setMessages(prev => [...prev, {
          role: 'ai',
          visual: { kind: 'map', payload: mapPayload }
      }])
  }

  但这段代码在 onDone 中执行，对话框模式应该能看到地图。

  3. 全屏模式无地图的根本原因

  // App.tsx 第350-356行 - 全屏模式渲染
  {moduleData.weather_analysis && (
      <ModuleCard
          module={moduleData.weather_analysis}  // ← visuals=[]
          amapKey={config?.amapPublicKey}
      />
  )}

  appendModuleResult 函数（第47-116行）只处理流式模块结果：
  // 只在收到 'result' 事件时调用
  onResult: (module: string, data: any) => {
      appendModuleResult(module, data)
  }

  但 upwind_enterprises 数据在 done 事件中，不会触发 appendModuleResult！
