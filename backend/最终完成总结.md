## 🔧 技术实现

### 核心代码结构
```python
class GetGuangdongRegularStationsTool(LLMTool):
    def __init__(self):
        # Schema定义：只接受question参数
        function_schema = {
            "name": "get_guangdong_regular_stations",
            "parameters": {
                "properties": {
                    "question": {
                        "type": "string",
                        "description": "自然语言查询问题..."
                    }
                },
                "required": ["question"]
            }
        }

    async def execute(self, question: str, **kwargs):
        # 1. 直接传递给UQP接口
        response = await http_client.post(
            "http://180.184.91.74:9091/api/uqp/query",
            json_data={
                "question": question,
                "intent_hint": "QUERY_STATION_INFO"
            }
        )

        # 2. 提取数据（参考VOCs工具）
        records = self._seek_records(response)

        # 3. 返回格式（参考VOCs工具）
        return {
            "success": True,
            "data": records,
            "count": len(records),
            "question": question,
            "summary": f"找到 {len(records)} 个监测站点",
        }
```
