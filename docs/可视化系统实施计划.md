# 可视化系统优化实施计划

## 项目概述

**目标**：构建基于业界最佳实践的智能可视化系统，支持：
- 声明式JSON规范（借鉴Vega-Lite）
- LLM驱动的智能图表生成（借鉴Microsoft LIDA）
- 自动质量评估与修复
- 丰富的专业图表类型（环境监测、气象等）

**核心架构**：6层设计
```
数据摘要层 → 数据准备层 → 策略选择层 → 规范生成层 → 评估修复层 → 前端渲染层
```

```python
"""
可视化规范 - 借鉴Vega-Lite的声明式设计
V1.0 - 2025-01

核心理念：
1. 声明式：描述"要什么"，而非"怎么做"
2. 安全性：JSON规范，无代码注入风险
3. 语言无关：前端可直接使用
4. 可存储：支持版本控制和回溯
"""

from __future__ import annotations
from typing import Union, List, Dict, Optional, Literal, Any
from pydantic import BaseModel, Field, validator
from datetime import datetime


# ============================================
# 基础类型定义
# ============================================

class DataReference(BaseModel):
    """数据引用"""
    data_id: str = Field(description="数据ID，格式：schema:version:hash")
    schema: str = Field(description="数据类型，如vocs、pmf_result、air_quality等")

    class Config:
        schema_extra = {
            "example": {
                "data_id": "vocs:v1:abc123",
                "schema": "vocs"
            }
        }


class EncodingChannel(BaseModel):
    """编码通道 - 数据字段到视觉属性的映射"""

    field: str = Field(description="数据字段名")

    type: Literal["quantitative", "nominal", "ordinal", "temporal"] = Field(
        description="字段类型：quantitative(数值)、nominal(类别)、ordinal(有序类别)、temporal(时间)"
    )

    aggregate: Optional[Literal["sum", "mean", "median", "min", "max", "count"]] = Field(
        default=None,
        description="聚合函数"
    )

    title: Optional[str] = Field(default=None, description="轴标题或图例标题")

    scale: Optional[Dict[str, Any]] = Field(
        default=None,
        description="刻度配置，如domain、range等"
    )

    axis: Optional[Dict[str, Any]] = Field(
        default=None,
        description="坐标轴配置，如grid、labels等"
    )

    legend: Optional[Dict[str, Any]] = Field(
        default=None,
        description="图例配置"
    )

    class Config:
        schema_extra = {
            "example": {
                "field": "PM2.5",
                "type": "quantitative",
                "title": "PM2.5浓度(μg/m³)",
                "scale": {"domain": [0, 100]}
            }
        }


class Transform(BaseModel):
    """数据转换"""

    type: Literal["filter", "aggregate", "bin", "calculate", "sort", "window"] = Field(
        description="转换类型"
    )

    params: Dict[str, Any] = Field(description="转换参数")

    class Config:
        schema_extra = {
            "example": {
                "type": "filter",
                "params": {"field": "PM2.5", "gt": 35}
            }
        }

# 可视化规范（核心）
# ============================================

class VisualizationSpec(BaseModel):
    """
    统一的可视化规范 - 类Vega-Lite设计

    核心概念：
    - mark: 图形标记（bar、line、point等）
    - encoding: 数据字段到视觉通道的映射
    - data: 数据引用
    - transform: 数据转换（可选）
    """

    # ========== 核心字段 ==========

    mark: Literal[
        "bar", "line", "point", "area", "pie",
        "timeseries", "scatter", "heatmap",
        "wind_rose", "profile", "map_trajectory",
        "boxplot", "radar"
    ] = Field(description="图形标记类型")

    data: Union[DataReference, Dict[str, Any]] = Field(
        description="数据引用（data_id）或内联数据"
    )

    encoding: Dict[str, EncodingChannel] = Field(
        description="编码映射：x、y、color、size等通道"
    )

    # ========== 可选字段 ==========

    transform: Optional[List[Transform]] = Field(
        default=None,
        description="数据转换流程"
    )

    title: Optional[str] = Field(default=None, description="图表标题")

    description: Optional[str] = Field(default=None, description="图表描述")

    width: Optional[int] = Field(default=None, description="图表宽度（像素）")

    height: Optional[int] = Field(default=None, description="图表高度（像素）")

    # ========== 交互配置 ==========

    selection: Optional[Dict[str, Any]] = Field(
        default=None,
        description="交互选择配置（如brush、interval等）"
    )

    # ========== 样式配置 ==========

    config: Optional[Dict[str, Any]] = Field(
        default=None,
        description="全局样式配置"
    )

    # ========== 元数据 ==========

    metadata: Optional[Dict[str, Any]] = Field(
        default=None,
        description="元数据，如生成方式、质量分数等"
    )

    @validator('encoding')
    def validate_encoding(cls, v, values):
        """验证编码配置的合理性"""
        mark = values.get('mark')

        # 基础图表需要x和y
        if mark in ['bar', 'line', 'point', 'area', 'scatter']:
            if 'x' not in v or 'y' not in v:
                raise ValueError(f"{mark}图表必须包含x和y编码")

        # 饼图需要角度或值
        if mark == 'pie':
            if 'theta' not in v and 'value' not in v:
                raise ValueError("饼图必须包含theta或value编码")

        return v

    class Config:
        schema_extra = {
            "example": {
                "mark": "line",
                "data": {
                    "data_id": "air_quality:v1:abc123",
                    "schema": "air_quality"
                },
                "encoding": {
                    "x": {
                        "field": "timePoint",
                        "type": "temporal",
                        "title": "时间"
                    },
                    "y": {
                        "field": "PM2.5",
                        "type": "quantitative",
                        "title": "PM2.5浓度(μg/m³)"
                    },
                    "color": {
                        "field": "station_name",
                        "type": "nominal",
                        "title": "站点"
                    }
                },
                "title": "各站点PM2.5时序变化",
                "width": 800,
                "height": 400
            }
        }

def spec_to_vegalite(spec: VisualizationSpec) -> Dict[str, Any]:
    """
    将VisualizationSpec转换为标准Vega-Lite格式

    用于前端直接使用Vega-Lite渲染器
    """
    vegalite_spec = {
        "$schema": "https://vega.github.io/schema/vega-lite/v5.json",
        "mark": spec.mark,
        "encoding": {}
    }

    # 转换encoding
    for channel, enc in spec.encoding.items():
        vegalite_spec["encoding"][channel] = {
            "field": enc.field,
            "type": enc.type
        }

        if enc.aggregate:
            vegalite_spec["encoding"][channel]["aggregate"] = enc.aggregate

        if enc.title:
            vegalite_spec["encoding"][channel]["title"] = enc.title

        if enc.scale:
            vegalite_spec["encoding"][channel]["scale"] = enc.scale

    # 添加可选字段
    if spec.title:
        vegalite_spec["title"] = spec.title

    if spec.width:
        vegalite_spec["width"] = spec.width

    if spec.height:
        vegalite_spec["height"] = spec.height

    # 处理数据引用
    if isinstance(spec.data, DataReference):
        vegalite_spec["data"] = {"name": spec.data.data_id}
    else:
        vegalite_spec["data"] = {"values": spec.data}

    return vegalite_spec
```

```python
def get_catalog_entry(schema: str) -> Optional[DataCatalogEntry]:
    """获取数据目录条目"""
    return CHART_DATA_CATALOG.get(schema)


def get_recommended_templates(schema: str, scenario: Optional[str] = None) -> List[ChartTemplate]:
    """
    获取推荐的图表模板

    Args:
        schema: 数据类型
        scenario: 使用场景（可选，用于过滤）

    Returns:
        按优先级排序的模板列表
    """
    entry = get_catalog_entry(schema)
    if not entry:
        return []

    templates = entry.recommended_templates

    # 如果指定场景，过滤匹配的模板
    if scenario:
        templates = [t for t in templates if scenario in t.scenario]

    # 按优先级排序
    templates.sort(key=lambda t: t.priority)

    return templates


def validate_data_fields(schema: str, data_fields: List[str]) -> Dict[str, Any]:
    """
    验证数据字段是否满足要求

    Returns:
        {
            "valid": True/False,
            "missing_fields": [...],
            "available_templates": [...]
        }
    """
    entry = get_catalog_entry(schema)
    if not entry:
        return {
            "valid": False,
            "error": f"未知的数据类型: {schema}"
        }

    # 检查必需字段
    missing_fields = [f for f in entry.required_fields if f not in data_fields]

    if missing_fields:
        return {
            "valid": False,
            "missing_fields": missing_fields,
            "message": f"缺少必需字段: {', '.join(missing_fields)}"
        }

    # 检查可用的模板（考虑条件）
    available_templates = []
    for template in entry.recommended_templates:
        # 检查条件
        if template.condition:
            # 简单的条件检查
            if template.condition == "has_timeseries" and "timeseries" not in data_fields:
                continue
            if template.condition == "has_pbl_data" and "pbl" not in data_fields:
                continue

        available_templates.append(template)

    return {
        "valid": True,
        "missing_fields": [],
        "available_templates": available_templates
    }
```

### 任务1.3：实现策略选择器

**文件**：`backend/app/tools/visualization/chart_strategy.py`

**目标**：智能选择图表生成策略（模板/推荐/LLM/Fallback）

**代码实现**：

```python
"""
图表生成策略选择器

优先级：
1. 精确匹配模板（速度快、质量稳定）
2. 智能推荐（基于数据特征）
3. LLM自定义生成（灵活但慢）
4. Fallback（兜底）
"""

from typing import Dict, Any, Optional, Literal
import structlog

from app.tools.visualization.chart_data_catalog import (
    get_catalog_entry,
    get_recommended_templates,
    validate_data_fields
)

logger = structlog.get_logger()

StrategyType = Literal["template", "intelligent_recommend", "llm_custom", "fallback"]


class ChartStrategySelector:
    """图表生成策略选择器"""

    def select_strategy(
        self,
        data_id: str,
        schema: str,
        user_intent: Optional[str] = None,
        prefer_custom: bool = False,
        data_summary: Optional[Dict[str, Any]] = None
    ) -> StrategyType:
        """
        选择图表生成策略

        Args:
            data_id: 数据ID
            schema: 数据类型
            user_intent: 用户意图（自然语言）
            prefer_custom: 用户是否偏好自定义
            data_summary: 数据摘要（来自DataSummarizer）

        Returns:
            策略类型
        """
        logger.info(
            "strategy_selection_start",
            schema=schema,
            has_user_intent=user_intent is not None,
            prefer_custom=prefer_custom
        )

        # 策略1：精确匹配模板
        template_strategy = self._try_template_strategy(schema, user_intent)
        if template_strategy:
            logger.info(
                "strategy_selected",
                strategy="template",
                reason=template_strategy["reason"]
            )
            return "template"

        # 策略2：智能推荐（基于数据特征）
        if not prefer_custom and data_summary:
            recommend_strategy = self._try_recommend_strategy(schema, data_summary, user_intent)
            if recommend_strategy:
                logger.info(
                    "strategy_selected",
                    strategy="intelligent_recommend",
                    reason=recommend_strategy["reason"]
                )
                return "intelligent_recommend"

        # 策略3：LLM自定义（用户明确需要或场景复杂）
        if prefer_custom or self._is_complex_intent(user_intent):
            logger.info(
                "strategy_selected",
                strategy="llm_custom",
                reason="用户需要自定义或场景复杂"
            )
            return "llm_custom"

        # 策略4：Fallback（兜底）
        logger.warning(
            "strategy_selected",
            strategy="fallback",
            reason="无合适策略，使用fallback"
        )
        return "fallback"

    def _try_template_strategy(
        self,
        schema: str,
        user_intent: Optional[str]
    ) -> Optional[Dict[str, str]]:
        """尝试模板策略"""
        # 检查catalog中是否有该类型
        entry = get_catalog_entry(schema)
        if not entry:
            return None

        # 如果没有用户意图，直接使用优先级最高的模板
        if not user_intent:
            return {
                "reason": "无用户意图，使用默认模板",
                "template": entry.recommended_templates[0]
            }

        # 如果用户意图与模板场景匹配，使用模板
        for template in entry.recommended_templates:
            if self._match_scenario(user_intent, template.scenario):
                return {
                    "reason": f"用户意图匹配模板场景: {template.scenario}",
                    "template": template
                }

        return None

    def _try_recommend_strategy(
        self,
        schema: str,
        data_summary: Dict[str, Any],
        user_intent: Optional[str]
    ) -> Optional[Dict[str, str]]:
        """尝试智能推荐策略"""
        # 基于数据特征判断
        statistics = data_summary.get("statistics", {})

        # 如果数据特征明确，可以推荐
        if statistics.get("has_time_series"):
            return {
                "reason": "数据包含时序信息，使用智能推荐"
            }

        if statistics.get("has_multiple_categories"):
            return {
                "reason": "数据包含多个分类，使用智能推荐"
            }

        return None

    def _match_scenario(self, user_intent: str, scenario: str) -> bool:
        """判断用户意图是否匹配场景"""
        # 关键词映射
        keywords_map = {
            "组分占比": ["占比", "比例", "分布", "饼图", "组成"],
            "浓度趋势": ["趋势", "变化", "时序", "时间", "走势"],
            "污染物对比": ["对比", "比较", "柱状图", "排名"],
            "物种对比": ["对比", "比较", "物种", "排名"],
            "风向玫瑰": ["风向", "风", "玫瑰", "风速"],
            "源贡献占比": ["源", "贡献", "占比", "来源"],
            "时空分布": ["时空", "分布", "热力图", "热力"],
            "站点对比": ["站点", "对比", "比较"],
            "多站点时序": ["多站点", "站点", "时序", "变化"],
        }

        keywords = keywords_map.get(scenario, [scenario.lower()])

        user_intent_lower = user_intent.lower()
        return any(kw in user_intent_lower for kw in keywords)

    def _is_complex_intent(self, user_intent: Optional[str]) -> bool:
        """判断是否为复杂意图（需要LLM处理）"""
        if not user_intent:
            return False

        # 包含复杂关键词
        complex_keywords = [
            "联动", "叠加", "多图", "组合", "对比分析",
            "相关性", "影响", "溯源", "综合", "多维度",
            "交叉", "关联", "因果"
        ]

        return any(kw in user_intent for kw in complex_keywords)

    def get_strategy_config(
        self,
        strategy: StrategyType,
        schema: str,
        user_intent: Optional[str] = None
    ) -> Dict[str, Any]:
        """
        获取策略的执行配置

        Returns:
            {
                "executor": "工具名称",
                "params": {...},
                "timeout": 秒数
            }
        """
        if strategy == "template":
            # 使用模板生成
            templates = get_recommended_templates(schema, user_intent)
            return {
                "executor": "generate_chart",
                "params": {
                    "use_template": True,
                    "template": templates[0] if templates else None
                },
                "timeout": 5
            }

        elif strategy == "intelligent_recommend":
            # 使用智能推荐
            return {
                "executor": "smart_chart_generator",
                "params": {
                    "auto_recommend": True
                },
                "timeout": 10
            }

        elif strategy == "llm_custom":
            # 使用LLM自定义
            return {
                "executor": "intelligent_visualization_planner",
                "params": {
                    "use_llm": True
                },
                "timeout": 30
            }

        else:  # fallback
            # 使用基础图表
            return {
                "executor": "generate_chart",
                "params": {
                    "chart_type": "bar",  # 默认柱状图
                    "simple_mode": True
                },
                "timeout": 5
            }
```

**验收标准**：
- ✅ 4种策略清晰定义
- ✅ 优先级逻辑正确
- ✅ 场景匹配准确（关键词映射）
- ✅ 复杂意图识别有效

```python
def _generate_guangdong_stations_timeseries(
    data: Union[List[Dict[str, Any]], List[Any]],

    for item in data:

            if not time_point:

            time_points.add(time_point)

            measurements = getattr(item, 'measurements', {})

            if station_name not in station_series:

                value = float(value) if value and value != "-" else None
            except (ValueError, TypeError):
                value = None

            if value is not None:
                station_series[station_name][time_point] = value

    if not time_points or not station_series:
        return {"error": "广东站点数据中缺少时间或站点信息"}

    # 排序时间点
    time_points_sorted = sorted(list(time_points))

    # 选择数据最完整的前10个站点
    station_completeness = {
        name: len([v for v in series.values() if v is not None])
        for name, series in station_series.items()
    }

        station_completeness.items(),

            "data": values
        })

        "series": series

            "pollutant": pollutant,

        }
    }

def _generate_guangdong_stations_pie(
    data: Union[List[Dict[str, Any]], List[Any]],
    pollutant: str = "PM2.5"
) -> Dict[str, Any]:
    """生成广东站点饼图（各站点平均浓度占比）"""
    # 计算各站点平均浓度
    station_totals = {}
    station_counts = {}

    for item in data:
        if isinstance(item, dict):
            station_name = item.get("station_name", "Unknown")
            measurements = item.get("measurements", {})

            if station_name not in station_totals:
                station_totals[station_name] = 0.0
                station_counts[station_name] = 0

            pollutant_key = pollutant.replace(".", "_")
            value = measurements.get(f"p{pollutant_key}") or measurements.get(pollutant)

            try:
                value = float(value) if value and value != "-" else None
                if value is not None and value > 0:
                    station_totals[station_name] += value
                    station_counts[station_name] += 1
            except (ValueError, TypeError):
                pass
        else:
            station_name = getattr(item, 'station_name', 'Unknown')
            measurements = getattr(item, 'measurements', {})

            if station_name not in station_totals:
                station_totals[station_name] = 0.0
                station_counts[station_name] = 0

            pollutant_key = pollutant.replace(".", "_")
            value = measurements.get(f"p{pollutant_key}") or measurements.get(pollutant)

            try:
                value = float(value) if value and value != "-" else None
                if value is not None and value > 0:
                    station_totals[station_name] += value
                    station_counts[station_name] += 1
            except (ValueError, TypeError):
                pass

    # 计算平均浓度
    station_averages = []
    for station_name in station_totals:
        if station_counts[station_name] > 0:
            avg = station_totals[station_name] / station_counts[station_name]
            station_averages.append({
                "name": station_name,
                "value": round(avg, 2)
            })

    # 排序并取前15个
    station_averages.sort(key=lambda x: x["value"], reverse=True)
    station_averages = station_averages[:15]

    return {
        "id": f"guangdong_stations_{pollutant}_pie",
        "type": "pie",
        "title": f"广东各站点{pollutant}平均浓度占比",
        "data": {
            "type": "pie",
            "data": station_averages
        },
        "meta": {
            "unit": "μg/m³",
            "data_source": "guangdong_stations",
            "record_count": len(data),
            "station_count": len(station_averages),
            "pollutant": pollutant
        }
    }

@staticmethod
def convert_guangdong_stations(
    data: Union[List[Dict[str, Any]], List[Any]],
    chart_type: str = "bar",
    **kwargs
) -> Dict[str, Any]:
    """
    将广东站点数据转换为图表数据

    Args:
        data: 广东站点数据
        chart_type: 图表类型（bar, timeseries, pie, line）
        **kwargs: 额外参数（pollutant等）

    Returns:
        转换后的图表数据
    """
    logger.info(
        "guangdong_stations_conversion_start",
        input_type=type(data).__name__,
        chart_type=chart_type,
        record_count=len(data) if isinstance(data, list) else 0
    )

    if not data:
        return {"error": "广东站点数据为空"}

    # 标准化图表类型
    chart_type = ChartDataConverter._normalize_chart_type(chart_type)
    pollutant = kwargs.get("pollutant", "PM2.5")

    if chart_type == "bar":
        return ChartDataConverter._generate_guangdong_stations_bar(data)
    elif chart_type == "timeseries":
        return ChartDataConverter._generate_guangdong_stations_timeseries(data, pollutant)
    elif chart_type == "pie":
        return ChartDataConverter._generate_guangdong_stations_pie(data, pollutant)
    elif chart_type == "line":
        # line图复用timeseries逻辑
        result = ChartDataConverter._generate_guangdong_stations_timeseries(data, pollutant)
        if "error" not in result:
            result["type"] = "line"
            result["data"]["type"] = "line"
        return result

    return {"error": f"不支持的广东站点图表类型: {chart_type}"}
```
