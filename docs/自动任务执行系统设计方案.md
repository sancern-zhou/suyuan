# 自动任务执行系统设计方案（优化版）

## 一、项目概述

### 1.1 设计目标

在大气污染溯源系统的现有 ReAct Agent 架构基础上，开发一个自动任务执行功能，实现：

- **用户交互**：用户只通过 Web UI（Vue 3）与 Agent 自然语言对话
- **任务调度**：支持 cron 表达式、固定间隔、一次性任务等多种调度方式
- **自动执行**：按照预设步骤自动完成分析任务
- **实时推送**：通过 WebSocket 实时推送执行状态和结果
- **结果追溯**：完整的执行历史记录和状态跟踪

### 1.2 设计理念

| 理念 | 说明 |
|------|------|
| **Web UI 优先** | 用户只通过浏览器界面交互，不使用 CLI 命令 |
| **Agent 服务化** | Agent 作为后端服务长期运行，非按需启动 |
| **轻量级调度** | 内置调度器（asyncio.sleep），无需第三方库 |
| **深度集成** | 复用现有 ReActLoop、ToolExecutor、DataContextManager |
| **渐进增强** | 从简单功能开始，逐步添加高级特性 |

### 1.3 参考项目借鉴

| 项目 | 借鉴特性 |
|------|----------|
| **OpenCLAW** | Gateway 守护进程架构、内置调度器（setTimeout）、任务持久化、多客户端支持 |
| **Claude Code** | 工具系统设计、Agent 按需启动模式 |

## 三、Gateway 服务设计

### 3.1 Gateway 架构

Gateway 是系统的**神经中枢**，负责：

| 职责 | 说明 |
|------|------|
| **单一连接点** | 所有客户端通过 WebSocket 连接到 Gateway |
| **消息路由** | 请求路由到对应处理器（Agent、Task、Session 等） |
| **事件分发** | 广播 Agent 执行、定时任务等事件到所有客户端 |
| **会话管理** | 维护多用户并发会话 |
| **状态同步** | 实时同步执行状态到所有连接的客户端 |

### 3.2 WebSocket 协议设计

```javascript
// 协议帧定义

// 请求帧 (客户端 → 服务器)
{
    "type": "req",
    "id": "unique_request_id",
    "method": "agent.run",          // 方法名
    "params": {                     // 参数
        "message": "分析广州污染数据"
    },
    "expectFinal": true              // 是否期待最终响应
}

// 响应帧 (服务器 → 客户端)
{
    "type": "res",
    "id": "unique_request_id",      // 对应请求ID
    "ok": true,                     // 是否成功
    "payload": {...},               // 响应数据
    "error": null                   // 错误信息（如果有）
}

// 事件帧 (服务器广播 → 所有客户端)
{
    "type": "event",
    "event": "agent",               // 事件名：agent, task, tick, etc.
    "payload": {...},               // 事件数据
    "seq": 123,                     // 序列号
    "timestamp": 17076544000000       // 时间戳
}
```

### 3.3 Gateway 核心实现

```python
# backend/app/core/gateway.py
from fastapi import WebSocket, WebSocketDisconnect
from typing import Set, Dict, Callable, Optional
import json
import asyncio
import logging
from datetime import datetime

logger = logging.getLogger(__name__)


class GatewayService:
    """
    Gateway 服务 - 系统神经中枢

    核心功能：
    1. WebSocket 连接管理
    2. 消息路由到对应处理器
    3. 事件广播到所有客户端
    4. 会话管理
    """

    def __init__(self):
        # 连接管理
        self.active_connections: Set[WebSocket] = set()
        self.client_info: Dict[WebSocket, Dict] = {}

        # 请求处理
        self.pending_requests: Dict[str, Callable] = {}

        # 事件处理器
        self.event_handlers: Dict[str, Callable] = {}

        # 序列号
        self._seq = 0

        # 会话管理
        self.sessions: Dict[str, Dict] = {}

    # ==================== 连接管理 ====================

    async def connect(self, websocket: WebSocket, client_id: str):
        """客户端连接"""
        await websocket.accept()

        self.active_connections.add(websocket)
        self.client_info[websocket] = {
            "client_id": client_id,
            "connected_at": datetime.now()
        }

        logger.info(f"Client connected: {client_id}, total: {len(self.active_connections)}")

        # 发送连接确认
        await self._send_event(websocket, "connected", {
            "clientId": client_id,
            "timestamp": int(datetime.now().timestamp() * 1000),
            "serverTime": datetime.now().isoformat()
        })

        # 广播连接事件
        await self.broadcast("presence", {
            "action": "connect",
            "clientId": client_id
        }, exclude=websocket)

    async def disconnect(self, websocket: WebSocket):
        """客户端断开"""
        client_info = self.client_info.get(websocket, {})
        client_id = client_info.get("client_id", "unknown")

        self.active_connections.discard(websocket)
        self.client_info.pop(websocket, None)

        logger.info(f"Client disconnected: {client_id}, total: {len(self.active_connections)}")

        # 广播断开事件
        await self.broadcast("presence", {
            "action": "disconnect",
            "clientId": client_id
        })

    # ==================== 消息处理 ====================

    async def handle_message(self, websocket: WebSocket, message: str):
        """处理客户端消息"""
        try:
            data = json.loads(message)

            if data["type"] == "req":
                # 处理请求
                await self._handle_request(websocket, data)
            elif data["type"] == "event":
                # 处理客户端事件（如心跳）
                await self._handle_client_event(websocket, data)
            else:
                await self._send_error(websocket, None, f"Unknown message type: {data['type']}")
                #           return

        except json.JSONDecodeError:
            await self._send_error(websocket, None, "Invalid JSON")
        except Exception as e:
            logger.error(f"Error handling message: {e}")
            await self._send_error(websocket, None, str(e))

    async def _handle_request(self, websocket: WebSocket, data: Dict):
        """处理请求"""
        req_id = data["id"]
        method = data["method"]
        params = data.get("params", {})

        try:
            # 路由到对应处理器
            handler = self._get_handler(method)

            if handler:
                # 调用处理器
                result = await handler(websocket, params)

                # 返回响应
                await self._send_response(websocket, req_id, {
                    "ok": True,
                    "payload": result
                })
            else:
                await self._send_error(websocket, req_id, f"Unknown method: {method}")

        except Exception as e:
            logger.error(f"Error handling request {method}: {e}")
            await self._send_error(websocket, req_id, str(e))

    def _get_handler(self, method: str) -> Optional[Callable]:
        """获取方法处理器"""
        # 方法路由
        if method.startswith("agent."):
            return self._agent_handler
        elif method.startswith("task."):
            return self._task_handler
        elif method == "health":
            return self._health_handler
        elif method == "sessions.list":
            return self._sessions_list_handler
        elif method == "sessions.get":
            return self._sessions_get_handler
        return None

    # ==================== 处理器实现 ====================

    async def _agent_handler(self, websocket: WebSocket, params: Dict):
        """Agent 请求处理器"""
        from app.main import get_react_loop

        message = params.get("message")
        if not message:
            raise ValueError("Missing 'message' parameter")

        react_loop = get_react_loop()

        # 流式执行 Agent
        final_result = None
        async for event in react_loop.run(message):
            # 实时推送到当前客户端
            await self._send_event(websocket, "agent", event)

            # 同时广播到所有客户端（可选）
            # await self.broadcast("agent", event, exclude=websocket)

            # 记录最终结果
            if event.get("type") == "final_answer":
                final_result = event.get("data", "")

        return {"status": "completed", "result": final_result}

    async def _task_handler(self, websocket: WebSocket, params: Dict):
        """任务请求处理器"""
        from app.main import get_task_scheduler

        scheduler = get_task_scheduler()
        action = params.get("action")

        if action == "create":
            # 创建任务
            task_id = await scheduler.create_task(**params)
            return {"taskId": task_id}

        elif action == "list":
            # 列出任务
            tasks = scheduler.list_tasks(enabled_only=params.get("enabledOnly", False))
            return {"tasks": [t.model_dump() for t in tasks]}

        elif action == "get":
            # 获取任务详情
            task = scheduler.get_task(params.get("taskId"))
            if not task:
                raise ValueError("Task not found")
            return task.model_dump()

        elif action == "update":
            # 更新任务
            task_id = params.get("taskId")
            success = await scheduler.update_task(task_id, **params.get("updates", {}))
            return {"success": success}

        elif action == "delete":
            # 删除任务
            task_id = params.get("taskId")
            success = await scheduler.delete_task(task_id)
            return {"success": success}

        elif action == "execute":
            # 执行任务
            task_id = params.get("taskId")
            execution = await scheduler.execute_task(task_id, manual_trigger=True)
            return execution.model_dump()

        elif action == "status":
            # 获取任务状态
            task_id = params.get("taskId")
            running = scheduler.get_running_task(task_id)
            if running:
                return {
                    "taskId": task_id,
                    "status": "running",
                    "executionId": running.execution_id,
                    "currentStep": running.current_step,
                    "startedAt": running.started_at.isoformat()
                }
            return {"taskId": task_id, "status": "idle"}

        elif action == "executions":
            # 获取执行历史
            task_id = params.get("taskId")
            limit = params.get("limit", 10)
            executions = scheduler.list_executions(task_id=task_id, limit=limit)
            return {"executions": [e.model_dump() for e in executions]}

        else:
            raise ValueError(f"Unknown action: {action}")

    async def _health_handler(self, websocket: WebSocket, params: Dict):
        """健康检查处理器"""
        return {
            "status": "ok",
            "connections": len(self.active_connections),
            "sessions": len(self.sessions),
            "timestamp": int(datetime.now().timestamp() * 1000)
        }

    async def _sessions_list_handler(self, websocket: WebSocket, params: Dict):
        """会话列表处理器"""
        return {
            "sessions": list(self.sessions.values())
        }

    async def _sessions_get_handler(self, websocket: WebSocket, params: Dict):
        """获取会话详情"""
        session_id = params.get("sessionId")
        session = self.sessions.get(session_id)
        if not session:
            raise ValueError("Session not found")
        return session

    # ==================== 事件处理 ====================

    async def _handle_client_event(self, websocket: WebSocket, data: Dict):
        """处理客户端事件（如心跳）"""
        event = data.get("event")

        if event == "ping":
            # 心跳响应
            await self._send_event(websocket, "pong", {
                "timestamp": int(datetime.now().timestamp() * 1000)
            })

    # ==================== 消息发送 ====================

    async def _send_response(self, websocket: WebSocket, req_id: str, payload: Dict):
        """发送响应"""
        message = {
            "type": "res",
            "id": req_id,
            **payload
        }
        await websocket.send_text(json.dumps(message))

    async def _send_error(self, websocket: WebSocket, req_id: Optional[str], error: str):
        """发送错误"""
        message = {
            "type": "res",
            "id": req_id,
            "ok": False,
            "error": error
        }
        await websocket.send_text(json.dumps(message))

    async def _send_event(self, websocket: WebSocket, event: str, payload: Dict):
        """发送事件到指定客户端"""
        self._seq += 1
        message = {
            "type": "event",
            "event": event,
            "seq": self._seq,
            "payload": payload,
            "timestamp": int(datetime.now().timestamp() * 1000)
        }
        await websocket.send_text(json.dumps(message))

    async def broadcast(self, event: str, payload: Dict, exclude: Optional[WebSocket] = None):
        """广播事件到所有客户端"""
        self._seq += 1
        message = {
            "type": "event",
            "event": event,
            "seq": self._seq,
            "payload": payload,
            "timestamp": int(datetime.now().timestamp() * 1000)
        }

        # 并发发送到所有连接
        tasks = []
        for connection in self.active_connections:
            if connection != exclude:
                tasks.append(self._safe_send(connection, message))

        if tasks:
            await asyncio.gather(*tasks, return_exceptions=True)

    async def _safe_send(self, websocket: WebSocket, message: Dict):
        """安全发送（处理断开的连接）"""
        try:
            await websocket.send_text(json.dumps(message))
        except Exception as e:
            # 连接已断开，移除
            logger.warning(f"Failed to send to client: {e}")
            self.active_connections.discard(websocket)

    # ==================== 会话管理 ====================

    def create_session(self, user_id: str, session_type: str = "analysis") -> str:
        """创建会话"""
        import uuid
        session_id = str(uuid.uuid4())

        self.sessions[session_id] = {
            "sessionId": session_id,
            "userId": user_id,
            "type": session_type,
            "createdAt": datetime.now().isoformat(),
            "status": "active"
        }

        return session_id

    def get_session(self, session_id: str) -> Optional[Dict]:
        """获取会话"""
        return self.sessions.get(session_id)

    def close_session(self, session_id: str):
        """关闭会话"""
        if session_id in self.sessions:
            self.sessions[session_id]["status"] = "closed"
            self.sessions[session_id]["closedAt"] = datetime.now().isoformat()


# 全局 Gateway 实例
gateway_service: GatewayService = None


def get_gateway() -> GatewayService:
    """依赖注入函数"""
```

## 五、内置调度器设计

### 5.1 调度器核心实现

```python
# backend/app/services/task_scheduler.py
import asyncio
import json
from pathlib import Path
from typing import Dict, List, Optional
from datetime import datetime, timedelta
import logging

from app.schemas.task import TaskTemplate, TaskExecution, TaskSchedule
from app.agent.core.task_executor import ScheduledTaskExecutor
from app.core.gateway import get_gateway

logger = logging.getLogger(__name__)


class TaskScheduler:
    """
    内置任务调度器 (借鉴 OpenCLAW)

    核心特性：
    1. 使用 asyncio.sleep 实现轻量级调度
    2. 单一定时器循环，基于最早任务时间
    3. 完整的任务生命周期管理
    4. 持久化和恢复
    5. 与 Gateway 集成，实时推送事件
    """

    # 常量
    MAX_TIMEOUT_MS = 2 ** 31 - 1  # 约24.8天
    STUCK_RUN_MS = 2 * 60 * 60 * 1000  # 2小时

    def __init__(
        self,
        react_loop_factory,
        storage_path: str = "backend_data_registry/tasks"
    ):
        self.react_loop_factory = react_loop_factory
        self.storage_path = Path(storage_path)
        self.storage_path.mkdir(parents=True, exist_ok=True)

        # 任务存储
        self.tasks: Dict[str, TaskTemplate] = {}
        self.executions: List[TaskExecution] = []
        self.running_tasks: Dict[str, TaskExecution] = {}

        # 调度器状态
        self.running = False
        self.timer_task = None
        self.lock = asyncio.Lock()

        # 加载已保存的任务
        self._load_tasks()

    # ==================== 生命周期管理 ====================

    def start(self):
        """启动调度器"""
        if not self.running:
            self.running = True
            self.timer_task = asyncio.create_task(self._timer_loop())
            logger.info("TaskScheduler started")

    def stop(self):
        """停止调度器"""
        if self.running:
            self.running = False
            if self.timer_task:
                self.timer_task.cancel()
            logger.info("TaskScheduler stopped")

    # ==================== 调度器循环 ====================

    async def _timer_loop(self):
        """
        定时器循环 (借鉴 OpenCLAW 的 armTimer + onTimer）

        工作流程：
        1. 计算下次唤醒时间（最早任务的下次执行时间）
        2. 等待到该时间
        3. 执行所有到期任务
        4. 重新计算并循环
        """
        while self.running:
            try:
                # 计算下次唤醒时间
                next_at = self._next_wake_at_ms()

                if next_at is None:
                    # 没有启用的任务，等待1分钟再检查
                    await asyncio.sleep(60)
                    continue

                # 计算延迟
                now_ms = self._now_ms()
                delay_ms = max(next_at - now_ms, 0)
                delay_sec = min(delay_ms / 1000, self.MAX_TIMEOUT_MS / 1000)

                logger.debug(f"Next wake at: {next_at}, delay: {delay_sec}s")

                # 等待到下次执行时间
                await asyncio.sleep(delay_sec)

                # 执行到期任务
                await self._run_due_jobs()

            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Timer loop error: {e}")
                await asyncio.sleep(60)  # 出错后等待1分钟再重试

    def _now_ms(self) -> int:
        """获取当前时间戳（毫秒）"""
        return int(datetime.now().timestamp() * 1000)

    def _next_wake_at_ms(self) -> Optional[int]:
        """计算下次唤醒时间（最早任务的下次执行时间）"""
        enabled = [
            task for task in self.tasks.values()
            if task.enabled and task.state.get("nextRunAtMs") is not None
        ]

        if not enabled:
            return None

        return min(
            task.state["nextRunAtMs"]
            for task in enabled
        )

    async def _run_due_jobs(self):
        """执行到期任务"""
        async with self.lock:
            now_ms = self._now_ms()

            # 找到所有到期任务
            due_tasks = [
                task for task in self.tasks.values()
                if task.enabled and task.state.get("nextRunAtMs", float('inf')) <= now_ms
            ]

            logger.info(f"Found {len(due_tasks)} due tasks")

            # 并发执行
            for task in due_tasks:
                # 检查是否已在运行
                if task.id in self.running_tasks:
                    logger.warning(f"Task {task.id} is already running, skipping")
                    continue

                # 在后台执行
                asyncio.create_task(self._execute_job(task))

    # ==================== 任务管理 ====================

    async def create_task(
        self,
        name: str,
        description: str,
        query: str,
        schedule: dict,
        enabled: bool = True,
        execution_mode: str = "isolated",
        metadata: dict = None
    ) -> str:
        """创建任务（简化版 - agent_message 模式）"""
        # 构建任务模板
        task = TaskTemplate(
            name=name,
            description=description,
            enabled=enabled,
            schedule=TaskSchedule(**schedule),
            execution_mode=execution_mode,
            steps=[
                {
                    "step_id": "main",
                    "name": "执行分析",
                    "executor": "agent_message",
                    "content": query
                }
            ],
            parameters=metadata or {}
        )

        # 计算下次执行时间
        task.state["nextRunAtMs"] = self._compute_next_run_at_ms(task)
        task.state["createdAtMs"] = self._now_ms()

        # 保存任务
        self.tasks[task.id] = task
        await self._persist_task(task)

        logger.info(f"Task created: {task.id} - {name}")
        return task.id

    async def update_task(self, task_id: str, **kwargs) -> bool:
        """更新任务"""
        task = self.tasks.get(task_id)
        if not task:
            return False

        # 更新字段
        for key, value in kwargs.items():
            if hasattr(task, key):
                setattr(task, key, value)

        # 重新计算下次执行时间
        if task.enabled:
            task.state["nextRunAtMs"] = self._compute_next_run_at_ms(task)

        # 持久化
        await self._persist_task(task)
        logger.info(f"Task updated: {task_id}")
        return True

    async def delete_task(self, task_id: str) -> bool:
        """删除任务"""
        if task_id not in self.tasks:
            return False

        del self.tasks[task_id]

        # 删除存储文件
        task_file = self.storage_path / f"{task_id}.json"
        if task_file.exists():
            task_file.unlink()

        logger.info(f"Task deleted: {task_id}")
        return True

    def get_task(self, task_id: str) -> Optional[TaskTemplate]:
        """获取任务"""
        return self.tasks.get(task_id)

    def list_tasks(self, enabled_only: bool = False) -> List[TaskTemplate]:
        """列出任务"""
        tasks = list(self.tasks.values())
        if enabled_only:
            tasks = [t for t in tasks if t.enabled]
        return tasks

    # ==================== 任务执行 ====================

    async def execute_task(
        self,
        task_id: str,
        manual_trigger: bool = False
    ) -> TaskExecution:
        """执行任务"""
        task = self.get_task(task_id)
        if not task:
            raise ValueError(f"Task {task_id} not found")

        # 检查是否已在运行
        if task_id in self.running_tasks:
            raise RuntimeError(f"Task {task_id} is already running")

        # 创建执行记录
        execution = TaskExecution(
            execution_id=f"exec_{self._now_ms()}",
            task_id=task_id,
            status="running",
            started_at=datetime.now()
        )
        self.executions.append(execution)
        self.running_tasks[task_id] = execution

        logger.info(f"Task execution started: {execution.execution_id}")

        # 通知 Gateway
        gateway = get_gateway()
        await gateway.broadcast("task", {
            "action": "started",
            "taskId": task_id,
            "executionId": execution.execution_id
        })

        try:
            # 使用 ScheduledTaskExecutor 执行
            react_loop = self.react_loop_factory()
            executor = ScheduledTaskExecutor(
                react_loop=react_loop,
                task=task,
                execution=execution,
                gateway=gateway
            )

            # 执行任务
            await executor.run()

            # 更新状态
            execution.status = "completed"
            execution.completed_at = datetime.now()
            execution.duration_ms = int(
                (execution.completed_at - execution.started_at).total_seconds() * 1000
            )
            execution.final_output = executor.get_summary()

            # 重新计算下次执行时间
            if task.enabled and not manual_trigger:
                task.state["nextRunAtMs"] = self._compute_next_run_at_ms(task, execution.completed_at)
                await self._persist_task(task)

        except Exception as e:
            logger.error(f"Task execution failed: {execution.execution_id}, error: {e}")
            execution.status = "failed"
            execution.completed_at = datetime.now()
            execution.error_message = str(e)

        finally:
            # 从运行中移除
            if task_id in self.running_tasks:
                del self.running_tasks[task_id]

            # 持久化执行记录
            await self._persist_execution(execution)

            # 通知 Gateway
            await gateway.broadcast("task", {
                "action": "finished",
                "taskId": task_id,
                "executionId": execution.execution_id,
                "status": execution.status,
                "result": execution.final_output
            })

        return execution

    def get_running_task(self, task_id: str) -> Optional[TaskExecution]:
        """获取正在运行的任务"""
        return self.running_tasks.get(task_id)

    def list_executions(
        self,
        task_id: str = None,
        limit: int = 50
    ) -> List[TaskExecution]:
        """获取执行历史"""
        executions = self.executions
        if task_id:
            executions = [e for e in executions if e.task_id == task_id]

        return sorted(executions, key=lambda x: x.started_at, reverse=True)[:limit]

    # ==================== 调度计算 ====================

    def _compute_next_run_at_ms(
        self,
        task: TaskTemplate,
        from_time: datetime = None
    ) -> Optional[int]:
        """计算下次执行时间（毫秒时间戳）"""
        from_time = from_time or datetime.now()

        if task.schedule.kind == "cron":
            # 使用 croniter 解析 cron 表达式
            try:
                from croniter import croniter
                cron = croniter(
                    task.schedule.expression,
                    from_time
                )
                next_time = cron.get_next(datetime)
                return int(next_time.timestamp() * 1000)
            except ImportError:
                logger.warning("croniter not installed, using simple cron parsing")
                # 简单实现（仅支持基础 cron）
                return self._parse_simple_cron(task.schedule.expression, from_time)

        elif task.schedule.kind == "interval":
            # 固定间隔
            interval_ms = task.schedule.expression
            from_ms = int(from_time.timestamp() * 1000)
            return from_ms + interval_ms

        elif task.schedule.kind == "once":
            # 一次性任务
            from datetime import isoformat
            return int(datetime.fromisoformat(task.schedule.expression).timestamp() * 1000)

        elif task.schedule.kind == "manual":
            # 手动触发，无下次执行时间
            return None

        return None

    def _parse_simple_cron(self, expression: str, from_time: datetime) -> Optional[int]:
        """简单 cron 解析（仅支持基础格式）"""
        # 简化实现，实际应使用 croniter
        parts = expression.split()
        if len(parts) != 5:
            return None

        minute, hour, day, month, weekday = parts

        # 简单处理：仅在整点执行
        if hour != "*" and minute == "0":
            next_time = from_time.replace(hour=int(hour), minute=0, second=0, microsecond=0)
            if next_time <= from_time:
                next_time += timedelta(days=1)
            return int(next_time.timestamp() * 1000)

        return None

    # ==================== 持久化 ====================

    async def _persist_task(self, task: TaskTemplate):
        """持久化任务"""
        task_file = self.storage_path / f"{task.id}.json"
        with open(task_file, "w", encoding="utf-8") as f:
            f.write(task.model_dump_json(indent=2))

        # 备份
        try:
            import shutil
            shutil.copy(task_file, f"{task_file}.bak")
        except:
            pass

    async def _persist_execution(self, execution: TaskExecution):
        """持久化执行记录"""
        exec_dir = self.storage_path / "executions"
        exec_dir.mkdir(exist_ok=True)

        exec_file = exec_dir / f"{execution.execution_id}.json"
        with open(exec_file, "w", encoding="utf-8") as f:
            f.write(execution.model_dump_json(indent=2))

    def _load_tasks(self):
        """加载已保存的任务"""
        for task_file in self.storage_path.glob("*.json"):
            if task_file.name == "registry.json" or task_file.name.endswith(".bak"):
                continue

            # 跳过执行记录目录
            if task_file.parent.name == "executions":
                continue

            try:
                with open(task_file, "r", encoding="utf-8") as f:
                    task_dict = json.load(f)
                    task = TaskTemplate(**task_dict)
                    self.tasks[task.id] = task

                    # 重新计算下次执行时间
                    if task.enabled:
                        task.state["nextRunAtMs"] = self._compute_next_run_at_ms(task)

            except Exception as e:
                logger.error(f"Failed to load task {task_file}: {e}")

        logger.info(f"Loaded {len(self.tasks)} tasks")
```

## 六、任务执行器设计

### 6.1 核心实现

```python
# backend/app/agent/core/task_executor.py
from typing import Dict, Any, List
import logging

from app.schemas.task import TaskTemplate, TaskStep, TaskExecution
from app.core.gateway import GatewayService

logger = logging.getLogger(__name__)


class ScheduledTaskExecutor:
    """
    定时任务执行器

    核心功能：
    1. 执行任务模板中的步骤
    2. 处理步骤依赖关系（拓扑排序）
    3. 支持并行执行
    4. 实时推送执行状态到 Gateway
    5. 完整的状态跟踪
    """

    def __init__(
        self,
        react_loop,
        task: TaskTemplate,
        execution: TaskExecution,
        gateway: GatewayService
    ):
        self.react_loop = react_loop
        self.task = task
        self.execution = execution
        self.gateway = gateway

        # 执行状态
        self.step_results: Dict[str, Dict[str, Any]] = {}
        self.data_ids: List[str] = []
        self.chart_ids: List[str] = []

    async def run(self):
        """运行任务"""
        logger.info(f"Starting task execution: {self.execution.execution_id}")

        # 构建步骤依赖图
        step_graph = self._build_step_graph()

        # 拓扑排序确定执行顺序
        sorted_steps = self._topological_sort(step_graph)

        # 按顺序执行步骤
        for step_id in sorted_steps:
            step = next((s for s in self.task.steps if s.step_id == step_id), None)
            if not step:
                continue

            # 检查是否应该跳过
            if self._should_skip_step(step):
                logger.info(f"Skipping step {step_id} due to dependency failure")
                continue

            # 更新当前步骤
            self.execution.current_step = step_id

            # 推送步骤开始事件
            await self.gateway.broadcast("task", {
                "action": "step_started",
                "taskId": self.task.id,
                "executionId": self.execution.execution_id,
                "stepId": step_id,
                "stepName": step.name
            })

            # 执行步骤
            result = await self._execute_step(step)
            self.step_results[step_id] = result

            # 推送步骤完成事件
            await self.gateway.broadcast("task", {
                "action": "step_finished",
                "taskId": self.task.id,
                "executionId": self.execution.execution_id,
                "stepId": step_id,
                "status": result.get("status", "unknown")
            })

            # 提取 data_id 和 chart_ids
            if "data_id" in result:
                self.data_ids.append(result["data_id"])
            if "chart_ids" in result:
                self.chart_ids.extend(result.get("chart_ids", []))

        # 更新执行记录
        self.execution.step_results = self.step_results
        self.execution.data_ids = self.data_ids
        self.execution.chart_ids = self.chart_ids

        logger.info(f"Task execution completed: {self.execution.execution_id}")

    async def _execute_step(self, step: TaskStep) -> Dict[str, Any]:
        """执行单个步骤"""
        logger.info(f"Executing step: {step.step_id} - {step.name}")

        try:
            if step.executor == "agent_message":
                return await self._execute_agent_message(step)

            elif step.executor == "direct_tool":
                return await self._execute_direct_tool(step)

            elif step.executor == "parallel":
                return await self._execute_parallel(step)

            else:
                raise ValueError(f"Unknown executor: {step.executor}")

        except Exception as e:
            logger.error(f"Step {step.step_id} failed: {e}")
            return {
                "status": "failed",
                "error": str(e)
            }

    async def _execute_agent_message(self, step: TaskStep) -> Dict[str, Any]:
        """通过 Agent 消息执行（复用 ReActLoop）"""
        result = {
            "status": "running",
            "data_id": None,
            "chart_ids": [],
            "summary": "",
            "tool_calls": []
        }

        # 使用现有的 ReActLoop 执行
        async for event in self.react_loop.run(step.content):

            # 推送执行事件到 Gateway
            await self.gateway.broadcast("agent", event)

            # 处理工具调用事件
            if event.get("type") == "tool_result":
                tool_data = event.get("data", {})

                # 记录工具调用
                if "generator" in event:
                    result["tool_calls"].append({
                        "tool": event["generator"],
                        "status": tool_data.get("status", "unknown")
                    })

                # 提取 data_id
                if isinstance(tool_data, dict) and "data_id" in tool_data:
                    result["data_id"] = tool_data["data_id"]

                # 提取 chart_ids
                if isinstance(tool_data, dict) and "visuals" in tool_data:
                    result["chart_ids"] = [
                        v.get("id") for v in tool_data["visuals"] if "id" in v
                    ]

            # 处理最终答案事件
            elif event.get("type") == "final_answer":
                result["summary"] = event.get("data", "")
                result["status"] = "completed"

        return result

    async def _execute_direct_tool(self, step: TaskStep) -> Dict[str, Any]:
        """直接调用工具"""
        from app.tools import global_tool_registry

        tool = global_tool_registry.get_tool(step.tool_name)
        if not tool:
            raise ValueError(f"Tool {step.tool_name} not found")

        # 调用工具
        result = await tool.execute(**(step.tool_params or {}))

        return {
            "status": "completed",
            "result": result
        }

    async def _execute_parallel(self, step: TaskStep) -> Dict[str, Any]:
        """并行执行子步骤"""
        import asyncio

        # 获取子步骤
        sub_steps = [
            s for s in self.task.steps if s.step_id in (step.sub_steps or [])
        ]

        # 并行执行
        results = await asyncio.gather(
            *[self._execute_step(s) for s in sub_steps],
            return_exceptions=True
        )

        return {
            "status": "completed",
            "sub_results": {
                s.step_id: r for s, r in zip(sub_steps, results)
            }
        }

    def _build_step_graph(self) -> Dict[str, List[str]]:
        """构建步骤依赖图"""
        return {
            s.step_id: s.depends_on or []
            for s in self.task.steps
        }

    def _topological_sort(self, graph: Dict[str, List[str]]) -> List[str]:
        """拓扑排序"""
        from collections import deque

        in_degree = {node: 0 for node in graph}
        for node in graph:
            for dep in graph[node]:
                if dep in in_degree:
                    in_degree[node] += 1

        queue = deque([node for node in in_degree if in_degree[node] == 0])
        result = []

        while queue:
            node = queue.popleft()
            result.append(node)

            for neighbor in graph:
                if node in graph[neighbor]:
                    in_degree[neighbor] -= 1
                    if in_degree[neighbor] == 0:
                        queue.append(neighbor)

        return result

    def _should_skip_step(self, step: TaskStep) -> bool:
        """检查是否应该跳过步骤"""
        if not step.depends_on:
            return False

        for dep_id in step.depends_on:
            dep_result = self.step_results.get(dep_id, {})
            if dep_result.get("status") == "failed":
                return True

        return False

    def get_summary(self) -> str:
        """生成执行摘要"""
        total = len(self.step_results)
        succeeded = sum(1 for r in self.step_results.values() if r.get("status") == "completed")

        lines = [
            f"任务执行完成：{succeeded}/{total} 步骤成功"
        ]

        if self.data_ids:
            lines.append(f"生成数据：{len(self.data_ids)} 份")

        if self.chart_ids:
            lines.append(f"生成图表：{len(self.chart_ids)}张")

        return "\n".join(lines)
```
